-- Team joining function using the exact logic from the provided scripts
local possibleRemotes = {
    "TeamChangeEvent",
    "TeamChanger",
    "ChangeTeamEvent",
    "TeamClient",
    "ChangeTeam",
    "JoinTeam"
}

local function joinTeam(teamName)
    local teamObj = game:GetService("Teams"):FindFirstChild(teamName)
    local success = false
    
    addLog("Attempting to join " .. teamName, "info")
    
    for _, remoteName in ipairs(possibleRemotes) do
        local remote = ReplicatedStorage:FindFirstChild(remoteName, true)
        
        if remote and remote:IsA("RemoteEvent") then
            -- Try all methods exactly as in the provided scripts
            pcall(function()
                remote:FireServer(teamName)
            end)
            
            if teamObj then
                pcall(function()
                    remote:FireServer(teamObj)
                end)
            end
            
            pcall(function()
                remote:FireServer("Join", teamName)
            end)
            
            pcall(function()
                remote:FireServer("Select", teamName)
            end)
            
            success = true
        end
        task.wait(0.05)
    end
    
    if success then
        addLog("Successfully joined " .. teamName, "success")
        
        -- Check if team actually changed (optional)
        task.wait(1)
        if Player.Team then
            addLog("Current team: " .. Player.Team.Name, "info")
        end
    else
        addLog("Failed to join team - no remotes found", "error")
    end
end

JoinBtn.MouseButton1Click:Connect(function()
    if selectedTeam == "MIT" then
        joinTeam("Milli İstihbarat Teşkilatı")
    elseif selectedTeam == "Askeri" then
        joinTeam("Askeri İnzibat")
    else
        addLog("Error: Please select a team first", "error")
    end
end)

-- Dragging functionality
local dragToggle, dragStart, startPos

TopBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragToggle = true
        dragStart = input.Position
        startPos = MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragToggle = false
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and dragToggle then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)
